81  /* Reads a 16-bit word. */
82  bool Read16(unsigned int* val) {
83  const char* buf;
84  if (!Read(&buf, 2)) return false;
85  unsigned int v = (*reinterpret_cast<const unsigned char*>(buf)) & 0xff;
86  ++buf;
87  v <<= 8;
88  v |= (*reinterpret_cast<const unsigned char*>(buf)) & 0xff;
89  *val = v;
90  return true;
91  }
